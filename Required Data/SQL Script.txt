CREATE TABLE  "GYMPLAN" 
   (	"PLANID" NUMBER, 
	"PLANNAME" VARCHAR2(60), 
	"PLANFEES" NUMBER, 
	"DURATIONINMONTHS" NUMBER, 
	 CONSTRAINT "GYMPLAN_PK" PRIMARY KEY ("PLANID") ENABLE
   )
/
CREATE TABLE  "MEMBER" 
   (	"MEMBERID" NUMBER NOT NULL ENABLE, 
	"NAME" VARCHAR2(60) NOT NULL ENABLE, 
	"ADDRESS" VARCHAR2(200), 
	"CONTACTNUMBER" VARCHAR2(67), 
	"BLOODGROUP" VARCHAR2(10), 
	"OCCUPATION" VARCHAR2(30), 
	"MEDICALHISTORY" VARCHAR2(1000), 
	"EMERGENCYCONTACTNO" VARCHAR2(67), 
	"IMAGE" BLOB, 
	"GENDER" VARCHAR2(10), 
	"IS_ACTIVE" NUMBER, 
	"WEIGHT" NUMBER(6,3), 
	"DATEOFBIRTH" DATE, 
	"REGISTRATIONDATE" DATE, 
	 CONSTRAINT "MEMBER_PK" PRIMARY KEY ("MEMBERID") ENABLE
   )
/
CREATE TABLE  "MEASUREMENTDETAILLS" 
   (	"MEMBERID" NUMBER, 
	"MEASUREMENTTAKENDATE" DATE, 
	"HEIGHT" NUMBER, 
	"WEIGHT" NUMBER, 
	"CHEST" NUMBER, 
	"WAIST" NUMBER, 
	"THIGHS" NUMBER, 
	"CALFS" NUMBER, 
	"ARMS" NUMBER, 
	"FOREARMS" NUMBER, 
	"FATINPER" NUMBER, 
	"BODYAGE" NUMBER, 
	"BMI" NUMBER, 
	"RM" NUMBER, 
	"VISCERALFAT" NUMBER, 
	"WHOLEBODYSF" NUMBER, 
	"WHOLEBODYSM" NUMBER, 
	"TRUNKSF" NUMBER, 
	"TRUNKSM" NUMBER, 
	"LEGSF" NUMBER, 
	"LEGSM" NUMBER, 
	"ARMSF" NUMBER, 
	"ARMSM" NUMBER, 
	 CONSTRAINT "MEASUREMENTDETAILLS_PK" PRIMARY KEY ("MEMBERID", "MEASUREMENTTAKENDATE") ENABLE,

   )
/
CREATE TABLE  "REGISTRATIONINFO" 
   (	"MEMBERID" NUMBER, 
	"PLANID" NUMBER, 
	"RECEIPTNO" NUMBER, 
	"STARTDATE" DATE, 
	"ENDDATE" DATE, 
	 CONSTRAINT "REGISTRATIONINFO_PK" PRIMARY KEY ("MEMBERID", "RECEIPTNO") ENABLE, 
	 CONSTRAINT "REGISTRATIONINFO_FK" FOREIGN KEY ("MEMBERID")
	  REFERENCES  "MEMBER" ("MEMBERID") ENABLE, 
	 CONSTRAINT "REGISTRATIONINFO_FK2" FOREIGN KEY ("PLANID")
	  REFERENCES  "GYMPLAN" ("PLANID") ENABLE
   )
/
